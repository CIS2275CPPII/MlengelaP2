#include<string>
using namespace std;


#ifndef SIMPLE_CALC_H
#define SIMPLE_CALC_H


class SimpleCalc
{
private:
	char operation{ '+' }; //which arithmetic operation is chosen
	double operand1{0.0};
	double operand2{1.0};//the answer in numerical form
	double answer{0}; 
	void Calculate();
	string desc; //description of the operation (optional)	
	string results; //formatted string

public:
	SimpleCalc(); // default constructor
	void SetOperation(char oper, double op1, double op2);
	string GetResults();

};


#endif // !SIMPLE_CALC_H


/*************************************************************************
* Program: Simple Calculator

#include <iostream>
#include <sstream>
#include <iomanip>
#include "SimpleCalc.h"

using namespace std;

void SimpleCalc::Calculate()
{
   string errMsg;

   switch (operation)
   {
      case '+':
      {
         answer = operand1 + operand2;
         desc   = "addition";
      }
      break;

      case '-':
      {
         answer = operand1 - operand2;
         desc   = "subtraction";
      }
      break;

      case '*':
      {
         answer = operand1 * operand2;
         desc   = "multiplication";
      }
      break;

      case '/':
      {
         if(operand2 == 0)
            errMsg = "Illegal Operation!";

         else
            answer = operand1 / operand2;

         desc = "division";
      }
      break;

      default:
      {
         results = "That's not a valid operator (";
         results += operation;
         results += "). Try again!";

         return;
      }
      break;
   }

   std::ostringstream out;

   out << "Your operation is " << desc << ": " <<
      operand1 << " " << operation << " " << operand2 << " ";

   if(errMsg != "")
      out << errMsg;

   else
      out << "= " << std::fixed << std::setprecision(2) << answer;

   results = out.str();
}

SimpleCalc::SimpleCalc() : operation('+'), operand1(0.0), operand2(0.0), answer(0.0)
{
}

void SimpleCalc::SetOperation(char oper, double op1, double op2)
{ 
   operation = oper;
   operand1  = op1;
   operand2  = op2;

   Calculate();
}

string SimpleCalc::GetResults()
{
   return(results);
}
